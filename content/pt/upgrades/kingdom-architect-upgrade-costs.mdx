  <Step num="3" title="Reinvista os Pagamentos em Melhorias">

  </Step>
  <Step num="4" title="Desbloqueie Categorias de Estruturas Avançadas">

  </Step>
  <Step num="5" title="Execute a Expansão de Território Controlada">
